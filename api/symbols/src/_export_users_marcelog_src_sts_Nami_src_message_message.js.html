 19 function Message() {
 20     this.lines = [];
 21     this.EOL = "\r\n";
 22 };
 23 
 24 Message.prototype.marshall = function () {
 25     var output = "";
 26     for (key in this) {
 27     	if (key == 'lines' || key == 'EOL' || (typeof this[key] === 'function')) {
 28     		continue;
 29     	}
 30     	output = output + key + ": " + this[key] + this.EOL;
 31     }
 32     output = output + this.EOL;
 33     return output;
 34 };
 35 
 36 Message.prototype.unmarshall = function (data) {
 37     this.lines = data.split(this.EOL);
 38     for (line in this.lines) {
 39         key = this.lines[line].split(":");
 40         /* XXX This should not be needed... */
 41         if (typeof(key[1]) === 'undefined') {
 42             continue;
 43         }
 44         this[key[0].replace(/-/, '_')] = key[1].replace(/^\s+/g, '').replace(/\s+$/g, '');
 45     }
 46 };
 47 
 48 Message.prototype.set = function(name, value) {
 49     this[name] = value;
 50 };
 51 
 52 Message.prototype.get = function(name) {
 53     return this[name];
 54 };
 55 
 56 exports.Message = Message;
